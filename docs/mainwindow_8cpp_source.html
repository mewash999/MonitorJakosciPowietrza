98void MainWindow::onDataReady(const QString &data) {
99 QJsonDocument doc = QJsonDocument::fromJson(data.toUtf8());
100 if (doc.isArray()) {
101 QJsonArray array = doc.array();
102 if (!array.isEmpty() && array.first().toObject().contains("stationName")) {
103 DataManager::saveHistoricalData("stations", data.toUtf8());
104 StationHandler::handleStationsData(array, ui->stationList, ui->lblStationCount, allStations);
105 } else {
106 DataManager::saveHistoricalData("sensors", data.toUtf8(), currentStationId);
107 SensorHandler::handleSensorsData(array, ui->sensorList, currentSensors);
108 }
109 } else if (doc.isObject()) {
110 DataManager::saveHistoricalData("measurements", data.toUtf8(), currentSensorId);
111
112 QJsonObject obj = doc.object();
113 QVector<QPair<QDateTime, double>> measurements;
114 QJsonArray values = obj["values"].toArray();
115 for (const QJsonValue &val : values) {
116 QJsonObject entry = val.toObject();
117 QDateTime date = QDateTime::fromString(entry["date"].toString(), Qt::ISODate);
118 double value = entry["value"].toDouble(-1.0);
119 if (value >= 0 && date.isValid()) {
120 measurements.append({date, value});
121 }
122 }
123 if (!measurements.isEmpty()) {
124 MeasurementHandler::handleMeasurementsData(obj, measurements, ui->lblStats);
125 MeasurementHandler::updateChart(measurements, ui->chartView, currentStationCity, currentStationAddress, currentParamName);
126 ui->lblStatus->setText("Wczytano dane online");
127 ui->lblStatus->setStyleSheet("color: green;");
128 } else {
129 ui->lblStatus->setText("Błąd danych online");
130 ui->lblStatus->setStyleSheet("color: red;");
131 }
132 }
133}

171void MainWindow::on_stationList_itemClicked(QListWidgetItem *item) {
172 ui->lblStats->clear();
173 currentStationId = item->data(Qt::UserRole).toInt();
174
175 auto stationData = std::find_if(allStations.begin(), allStations.end(),
176 [this](const QPair<QString, QJsonObject>& pair) {
177 return pair.second["id"].toInt() == currentStationId;
178 });
179

181 currentStationCity = stationData->second["city"].toObject()["name"].toString();

195 if (doc.isArray()) {

198 } else {
199 apiClient->fetchData(QUrl(QString("https://api.gios.gov.pl/pjp-api/rest/station/sensors/%1")

248 }

252 lblStatus->setStyleSheet("color: red;");

328void MainWindow::loadHistoricalData(int days) {
329 if (currentSensorId == -1) {
330 lblStatus->setText("Proszę wybrać stację oraz czujnik");
331 lblStatus->setStyleSheet("color: orange;");
332 return;
333 }
334
335 QVector<QPair<QDateTime, QByteArray>> allData = DataManager::loadAllHistoricalData("measurements", currentSensorId);
336 QMap<QDateTime, double> aggregatedData;
337
338 QDateTime cutoffDate = QDateTime::currentDateTime().addDays(-days);
339
340 for (const auto &dataPair : allData) {
341 QJsonDocument doc = QJsonDocument::fromJson(dataPair.second);
342 if (doc.isObject()) {
343 QJsonArray values = doc.object()["values"].toArray();
344 for (const QJsonValue &val : values) {
345 QJsonObject entry = val.toObject();
346 QDateTime date = QDateTime::fromString(entry["date"].toString(), Qt::ISODate);
347 if (date.isValid() && date >= cutoffDate) {
348 double value = entry["value"].toDouble(-1.0);
349 if (value >= 0) {
350 aggregatedData.insert(date, value);
351 }
352 }
353 }
354 }
355 }
356
357 QVector<QPair<QDateTime, double>> filteredData;
358 for (auto it = aggregatedData.constBegin(); it != aggregatedData.constEnd(); ++it) {
359 filteredData.append(qMakePair(it.key(), it.value()));
360 }
361
362 std::sort(filteredData.begin(), filteredData.end(),
363 [](const auto &a, const auto &b) { return a.first < b.first; });
364
365 if (!filteredData.isEmpty()) {
366 MeasurementHandler::handleMeasurementsData(QJsonObject(), filteredData, ui->lblStats);
367 MeasurementHandler::updateChart(filteredData, ui->chartView, currentStationCity, currentStationAddress, currentParamName);
368 lblStatus->setText(QString("Wczytano dane historyczne z %1 dni").arg(days));
369 lblStatus->setStyleSheet("color: orange;");
370 } else {
371 lblStatus->setText(QString("Brak danych historycznych z %1 dni").arg(days));
372 lblStatus->setStyleSheet("color: red;");
373 }
374}

444bool MainWindow::eventFilter(QObject *obj, QEvent *event) {
445 if (event->type() == QEvent::KeyPress) {
446 QKeyEvent *keyEvent = static_cast<QKeyEvent *>(event);
447 if (keyEvent->key() == Qt::Key_Up || keyEvent->key() == Qt::Key_Down || keyEvent->key() == Qt::Key_Left || keyEvent->key() == Qt::Key_Right || keyEvent->key() == Qt::Key_Space) {
448 if (obj == ui->btnLast7Days || obj == ui->btnLast14Days || obj == ui->btnHistory) {
449 return true;
450 }
451 }
452 }
453 return QMainWindow::eventFilter(obj, event);
