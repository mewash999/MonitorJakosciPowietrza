121QVector<QPair<QDateTime, QByteArray>> DataManager::loadAllHistoricalData(const QString &type, int id) {
122 QVector<QPair<QDateTime, QByteArray>> result;
123 QString path = QStandardPaths::writableLocation(QStandardPaths::AppDataLocation);
124 QDir dir(path);
125
126 if (type == "measurements") {
127 QString filter = QString("%1_%2_*.json").arg(type).arg(id);
128 QStringList files = dir.entryList(QStringList() << filter, QDir::Files, QDir::Time);
129
130 foreach (const QString &file, files) {
131 QFile f(dir.filePath(file));
132 if (f.open(QIODevice::ReadOnly)) {
133 QStringList parts = file.split('_');
134 if (parts.size() >= 4) {
135 QString timestampStr = parts[2] + "_" + parts[3].left(6);
136 QDateTime dt = QDateTime::fromString(timestampStr, "yyyyMMdd_HHmmss");
137 result.append(qMakePair(dt, f.readAll()));
138 }
139 f.close();
140 }
141 }
142 }
143 return result;
144}
